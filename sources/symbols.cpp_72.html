/**
 Symbol type to text translation.
 */
 
#include <QtCore/QString>
 
#include "symbols.h"
 
namespace Bww {
 
  static const char* symTable[] =
  {
    "COMMENT",
    "HEADER",
    "STRING",
    "CLEF",
    "KEY",
    "TEMPO",
    "TSIG",
    "PART",
    "BAR",
    "NOTE",
    "TIE",
    "TRIPLET",
    "DOT",
    "GRACE",
    "UNKNOWN",
    "NONE"
  };
 
  QString symbolToString(Symbol s)
  {
    if (s < 0)
      return "INVALID";
    if (static_cast<unsigned>(s) > sizeof symTable)
      return "INVALID";
    else
      return symTable[s];
  }
 
} // namespace Bww
