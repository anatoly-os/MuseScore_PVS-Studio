#include "fifo.h"
 
namespace Ms {
 
//---------------------------------------------------------
//   clear
//---------------------------------------------------------
 
void FifoBase::clear()
      {
      ridx    = 0;
      widx    = 0;
      counter = 0;
      }
 
//---------------------------------------------------------
//   push
//---------------------------------------------------------
 
void FifoBase::push()
      {
      widx = (widx + 1) % maxCount;
//      q_atomic_increment(&counter);
      ++counter;
      }
 
//---------------------------------------------------------
//   pop
//---------------------------------------------------------
 
void FifoBase::pop()
      {
      ridx = (ridx + 1) % maxCount;
      // q_atomic_decrement(&counter);
      --counter;
      }
 
}
