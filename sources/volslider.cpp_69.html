#include "fastlog.h"
#include "volslider.h"
 
namespace Awl {
 
//---------------------------------------------------------
//   VolSlider
//---------------------------------------------------------
 
VolSlider::VolSlider(QWidget* parent)
   : Slider(parent)
      {
      setLog(true);
//      setRange(-60.0f, 10.0f);
      setRange(-60.0f, 20.0f);
      setScaleWidth(7);
      setLineStep(.8f);
      setPageStep(3.0f);
      setDclickValue1(0.0);
      setDclickValue2(0.0);
      }
 
//---------------------------------------------------------
//   setValue
//---------------------------------------------------------
 
void VolSlider::setValue(double val)
      {
      if (_log) {
            if (val == 0.0f)
                  _value = _minValue;
            else {
                  _value = fast_log10(val) * 20.0f;
                  if (_value < _minValue)
                        _value = _minValue;
                  }
            }
      else
            _value = val;
      update();
      }
 
//---------------------------------------------------------
//   value
//---------------------------------------------------------
 
double VolSlider::value() const
      {
      double val = _log ? (_value <= _minValue) ? 0.0f : pow(10.0, _value*0.05f)
         : _value;
      return val;
      }
 
}
